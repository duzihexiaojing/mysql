use gouwuche;

drop table if exists yonghu;

create table yonghu(

yonghu_id smallint unsigned not null auto_increment primary key,

yonghu_name varchar(40) not null,

zhuangbei_id smallint unsigned not null,

last_update timestamp not null default current_timestamp on update current_timestamp,

key `idx_fk_zhuangbei_id` (zhuangbei_id),

constraint `fk_city_zhuangbei` foreign key(zhuangbei_id)

references zhuangbei(zhuangbei_id) 

on delete restrict on update cascade

)engine=innodb default charset=utf8;



use gouwuche;

drop table if exists zhuangbei;

create table zhuangbei(

zhuangbei_id smallint unsigned not null auto_increment,

zhuangbei_name varchar(40) not null,

money int unsigned not null,

last_update timestamp default current_timestamp on update current_timestamp,

primary key(zhuangbei_id)

)engine=innodb default charset=utf8;
